${(() => { const page = Number(data?.page || 1); const postsPerPage = 10; const offset = (page - 1) * postsPerPage; // Get posts from database const posts = engine.db.prepare(` SELECT p.*, json_extract(p.metadata, '$.date') as post_date, json_extract(p.metadata, '$.excerpt') as excerpt FROM pages p WHERE p.template = 'blog' ORDER BY post_date DESC LIMIT ? OFFSET ? `).all(postsPerPage, offset); // Check if there are more posts const hasMore = engine.db.prepare(` SELECT COUNT(*) as count FROM pages WHERE template = 'blog' `).get().count > (offset + postsPerPage); return engine.render('base', { ...data, content: `
    ${posts.length > 0 ? posts.map(post => `

  • ${post.title}

    ${post.post_date ? `${post.post_date}` : ''} ${post.excerpt ? `

    ${post.excerpt}

    ` : ''}
    • `).join('') : '
  • No posts available
    • '}
${page > 1 ? `← Previous` : '' } ${hasMore ? `Next →` : '' }
` }); })()}
